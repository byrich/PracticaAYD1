public int suma(int a, int b){
	return a + b;
}


public int multiplicacion(int a, int b){
	int resultado = a * b;
	return resultado;
}

public int resta(int a, int b){
	return a - b;
}

public int div(int a, int b){
	return a / b;
}


public int raiz(int a, int b){
	return Math.sqrt(a,b);

public int potencia(int a, int b){
	return a ^ b;
}
