public int suma(int a, int b){
	return a + b;
}


public int multiplicacion(int a, int b){
	int resultado = a * b;
	return resultado;
}

public int resta(int a, int b){
	return a - b;
}

public int div(int a, int b){
	if (b==0) {
		return 0;
	}
	return a / b;
}
